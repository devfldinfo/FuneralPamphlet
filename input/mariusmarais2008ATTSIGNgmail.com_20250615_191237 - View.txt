\begin{hymn}{6}{As my hart so swaar}

Laat ek in sy hande staan.
\end{stanza}

\end{chorus}
\begin{stanza}

\begin{hymn}{7}{Daar Is Geen Vriend Soos Jesus}

My eerste liefgehad.
\end{stanza}

My eerste liefgehad.

My eerste liefgehad.
\end{stanza}
\end{hymn}
