\begin{hymn}{42}{There is a way}{}
\begin{stanza}
There is a way, a narrow way,\\
That leads to life above -\\
A way of peace and holiness,\\
Of purity and love.
\end{stanza}
\begin{chorus}
\setchorusline{My yearning soul desires to find}
My yearning soul desires to find\\
This hidden path of bliss;\\
Lord, hear my prayer, and lead me in\\
The way of holiness.
\end{chorus}
\begin{stanza}
To find this way, this living way,\\
We must forego our sin:\\
God has declared the pure in heart\\
Alone can enter in.
\end{stanza}
\begin{stanza}
There is a way, a humble way,\\
A way of truth and grace -\\
The highway of God's righteousness,\\
Where self can have no place.
\end{stanza}
\begin{stanza}
There is a way, a perfect way;\\
His chosen walk therein;\\
And they who share His suffering now\\
Eternal glory win.
\end{stanza}
\end{hymn}

\begin{hymn}{53}{God has always laboured}{\#35}
\begin{stanza}
Lesa alafwayo kutupususha\\
Nga twafuma mu nshila isha membu;\\
Takuli na cimbi icho afwaya\\
Kano Kristu aikala muli 'fwe.
\end{stanza}
\begin{chorus}
\setchorusline{Kristu muli 'fwe, pusukilo;}
Kristu muli 'fwe, pusukilo;\\
Atucimfisha pa lubembu,\\
Twenda mu makasa yakwe pe,\\
Amonekwa mu micitile yesu.
\end{chorus}
\begin{stanza}
Bambi bafwayo kuilungamika\\
Ne misango yabo fye kuli Lesa;\\
Abo abashafyalwa cipya cipya\\
Tabakatale bengile mu mulu.
\end{stanza}
\begin{stanza}
Pelo mweo obe kuli Shikulu;\\
Sale nshila yakwe iyo - endamo;\\
Mwitina kube nte no kumukonka;\\
Alakupa maka nga wamubela.
\end{stanza}
\end{hymn}
