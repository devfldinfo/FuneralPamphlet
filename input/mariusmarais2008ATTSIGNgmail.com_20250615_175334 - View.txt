\begin{hymn}{234}{Bwana giza la jioni}
\hymntitlewithdropcap{234}{Bwana giza la jioni}
\begin{stanza}
Bwana giza la jioni lafika,\\
Laiti siku iwe ndefu zaidi,\\
Mavuno yamekwisha iva sana,\\
Ndiyo kwanza naanza kuvuna,\\
Bali ukiniita nije kwako,\\
Kabla kazi hii haijakwisha\\
Au ikiwa tumeshavuna yote,\\
Nisaidie nikujibu, ``Nipo Bwana.''
\end{stanza}
\begin{chorus}
\setchorusline{Ukirudi usiku wa manane,}
Ukirudi usiku wa manane,\\
Hata ukungu wa [al]fajiri,\\
Kusimamia utawala wako,\\
Bwana wangu, mpendwa, unikumbuke
\end{chorus}
\begin{stanza}
Ee! wengine walikutumikia,\\
Kwa mito walitumia mawe,\\
Wengine meli zao zilivunjika,\\
Hawakuhofu shida yo yote,\\
Nipake mafuta nifanye kazi,\\
Pamoja nao mashujaa hawa,\\
Nisaidie nivune mashambani\\
Nilete mavuno ghalani mwako juu.
\end{stanza}
\begin{stanza}
Niko tayari kwenda kuhubiri,\\
Vema zaidi nitumike kwako,\\
Natamani nafasi japo ndogo,\\
Napenda kutimiza wajibu,\\
Nitayashika mausia yako,\\
Hadi mwisho wa maisha yangu,\\
Nikilala kifoni, urudipo,\\
Ee! Bwana ni[am]she niwe kwako daima.
\end{stanza}
\end{hymn}
