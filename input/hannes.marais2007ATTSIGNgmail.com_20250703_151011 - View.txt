\begin{hymn}{236}{Baba nipe moyo}{}
\begin{stanza}
Baba, nipe moyo laini,\\
Wenye huruma na mnyofu.\\
Na ufanane na wako tu,\\
Kwa kila neno na tendo.\\
Ondoa ugumu,\\
Na uchungu pia.\\
Moyo mpya wenye\\
Ukarimu wako.\\
Nirudishe kwenye\\
Mapenzi yako tu.\\
Nipe moyo laini\\
Kama wako.
\end{stanza}
\begin{stanza}
Ee Baba, nipe moyo safi,\\
Ingawa sikustahili.\\
Nijaze Roho wako, Baba,\\
Nikupendeze daima.\\
Niyasahau ma-\\
kosa ya wengine.\\
Nikiwasamehe,\\
Kabisa moyoni\\
Ingawa machozi,\\
Ya huzuni yaje;\\
Nipe moyo laini\\
Kama wako.
\end{stanza}
\begin{stanza}
Ee Nipe moyo bila hila\\
Nijaze upendo safi.\\
Nipe moyo kumsaidia\\
Mwingine mwenye haja kuu.\\
Japo mpendwa au\\
Mwenye ukatili.\\
Nione huzuni\\
Ya jirani yangu.\\
Nishiriki taabu\\
Au maumivu;\\
Mwenye moyo laini\\
Niendako.
\end{stanza}
\end{hymn}
