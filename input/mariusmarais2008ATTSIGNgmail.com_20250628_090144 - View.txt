\begin{hymn}{20}{From heaven's glory}
\hymntitlewithdropcap{20}{From heaven's glory}
\begin{stanza}
From heaven's glory,\\
From His radiant throne above,\\
Came our Redeemer\\
In His wondrous love.\\
Oh, what pain and sorrow\\
Jesus suffered on the tree\\
As He died for sinners,\\
Died for you and me.
\end{stanza}
\begin{chorus}
\setchorusline{Saviour, my Saviour,}
Saviour, my Saviour,\\
Thou hast died to make me free;\\
Help me to serve Thee\\
Till Thy blessèd face I see.
\end{chorus}
\begin{stanza}
Faint and forsaken,\\
Out on Calvary's mountainside,\\
Jesus, my Saviour,\\
Bowed His head and died.\\
He, our hope and surety,\\
All this suffering meekly bore,\\
But He rose triumphant,\\
Liveth evermore.
\end{stanza}
\begin{stanza}
How can we grieve Him -\\
Jesus, blessèd Son of God -\\
Him, who so freely\\
Gave His precious blood?\\
With no hand to help us,\\
Drifting to a hopeless grave,\\
Jesus paid the ransom:\\
He alone could save.
\end{stanza}
\begin{stanza}
Come to the Saviour,\\
With thy weary load of care;\\
Tell Him thy sorrow;\\
He will hear thy prayer.\\
He will loose thy burden,\\
Make the victor over sin;\\
He will fill with gladness\\
All thy heart within.
\end{stanza}
\end{hymn}

\begin{hymn}{30}{We have found him}
\hymntitlewithdropcap{30}{We have found him}
\begin{stanza}
Ye who trace with weary hearts and sad\\
Those blest scenes of Galilee,\\
O rejoice and be forever glad!\\
"We have found Him - come and see!"
\end{stanza}
\begin{chorus}
\setchorusline{"We have found Him! - Joy of the Ages!"}
"We have found Him! - Joy of the Ages!"\\
And our song with heaven's gladness rings:\\
"We have found the Christ of whom the prophets spake;\\
We have found Him, King of kings."
\end{chorus}
\begin{stanza}
Ye who hunger for the living word,\\
Ye who thirst for living springs,\\
Come, each waiting heart with joy is stirred\\
By the song the herald sings.
\end{stanza}
\begin{stanza}
This same Christ who taught beside the sea\\
Walks upon the earth today,\\
And He comes in lowliness to thee,\\
Templed still in mortal clay.
\end{stanza}
\begin{stanza}
"We have found Him!" Bear the tidings far,\\
Wheresoever men are found,\\
Until all who seek the Guiding Star\\
Shall in light and peace abound.
\end{stanza}
\end{hymn}

\begin{hymn}{101}{Hier in u Naam, O Heer}
\hymntitlewithdropcap{101}{Hier in u Naam, O Heer...\#172}
\begin{stanza}
Hier in u Naam, O Heer, vandag\\
Ontmoet ons ned'rig, bly;\\
Help ons te spreek en bid en wag\\
U seën te verkry.
\end{stanza}
\begin{chorus}
\setchorusline{O Heer, laat nou u seën ons bestraal;}
O Heer, laat nou u seën ons bestraal;\\
In harte wat nou biddend wag\\
Mag volle seënstrome ryklik daal.
\end{chorus}
\begin{stanza}
Alleen met U, o salig' tyd,\\
Geen sorge en geen vrees;\\
Ootmoedig kom ons toegewyd\\
Biddend in hart en gees.
\end{stanza}
\begin{stanza}
Alleen met U, behoefte voel\\
Ons, Jesus, aan u bloed.\\
Reinig ons hart, gedagte, doel\\
Wil Self ons siele voed.
\end{stanza}
\begin{stanza}
O Heer, vandag mag daar weer styg\\
n Nuwe lied so soet,\\
Van lof en prys wat kan getuig\\
Van redding deur u bloed.
\end{stanza}
\end{hymn}

\begin{hymn}{200}{Tungependa kumwona}
\hymntitlewithdropcap{200}{Tungependa kumwona...\#392}
\begin{stanza}
Tungependa kumwona Yesu kwani,\\
Maisha yetu yapita hima,\\
Tumwone ili imani izidi,\\
Na tuwe thabiti hadi mwisho.
\end{stanza}
\begin{stanza}
Tumwone Yesu, msingi wa mwamba,\\
Ambao twasimama juu yake,\\
Usumbufu wa maisha na kifo\\
Hautatutoa tukimwona.
\end{stanza}
\begin{stanza}
Tumwone Yesu, kwa kuwa vingine\\
Tulivyopenda vinafifia.\\
Baraka za dunia zinapita\\
Haidhuru kwetu, twakujia.
\end{stanza}
\begin{stanza}
Tumwone Yesu ili tuzidishwe\\
Furaha, nguvu na utayari.\\
Tumwone akituombea sasa,\\
Ili tusikihofie kifo.
\end{stanza}
\end{hymn}

\begin{hymn}{10}{Is there anyone can help us}
\hymntitlewithdropcap{10}{Is there anyone can help us...\#47}

Ngubani ongasisiza, oqonda inhliziyo,\\
Laph' ameva ehlaba kuze kophe,\\
Yena osihawukela, asinike ngothando\\
Isibusiso esisidingayo?
\end{stanza}

\setchorusline{NguJesu, oyedwa,}

Othandeka kuYise, nguYena!\\
Lapho izinkathazo\\
Nohlupho lufika,\\
UMhlobo othembekile, nguJesu.
\end{chorus}
\begin{stanza}
Ngubani ongasisiza, aphumuze isoni,\\

\end{stanza}
\begin{stanza}

Ozohamba nathi ebumnyameni,\\

Samukelwe ekhaya naphakade?
\end{stanza}
\end{hymn}
