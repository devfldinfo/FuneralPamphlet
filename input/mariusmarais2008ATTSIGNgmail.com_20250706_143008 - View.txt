\begin{hymn}{53}{Ek kan nie verder sien}{\#277}
\begin{stanza}
Ek kan nie verder sien as hede.\\
Vir more kry 'k\textasciitilde{}nie krag vandag,\\
Maar, liewe Heer, bewaar my veilig\\
Vir slegs die een stap weer wat wag.
\end{stanza}
\begin{chorus}
\setchorusline{O Jesus, hou my elke skrede}
O Jesus, hou my elke skrede\\
Op weë wat U wys vir my.\\
Hou U my hand, Heer alvermoënd,\\
U sal my Hoop, my Sterkte bly.
\end{chorus}
\begin{stanza}
U het belowe in versoeking,\\
Gena' te skenk om te oorwin.\\
U maak voorsiening vir ontkoming\\
Uit elke strik so slu versin.
\end{stanza}
\begin{stanza}
Vergetend al die vele foute,\\
Die sonde wat my soms laat ween,\\
Met oog gerig nou reg na vore\\
Wens ek te volg slegs U alleen.
\end{stanza}
\begin{stanza}
Die storme wat my soms oorweldig,\\
My voetstap selfs verberg van my.\\
Geloof in God vrees vir geen duister\\
En Hy die stap tereg sal lei.
\end{stanza}
\end{hymn}
