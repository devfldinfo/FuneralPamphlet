\begin{hymn}{10}{Yesu alikufa}{Yesu alikufa}
\begin{stanza}
Yesu alikufa kwa wenye dhambi,\\
Acha dhambi zako umwone Mungu;\\
Patana na Yesu, utakuwa hai,\\
Kristo ndani ni taraja la utukufu.
\end{stanza}
\begin{chorus}
\setchorusline{Umkiri na mwamini, Yesu, Mwokozi,}
Umkiri na mwamini, Yesu, Mwokozi,\\
Roho asitoke kwako daima.
\end{chorus}
\begin{stanza}
Akuita wewe, mwenye mizigo,\\
Atakupokea akupumzishe;\\
Muda wa rehema, ziache dhambi,\\
Mungu atakusamehe akupe raha.
\end{stanza}
\begin{stanza}
Ukisamehewa kwa kifo chake,\\
Utaokolewa kwa uhai wake;\\
Akaavyo mwako usiogope,\\
Utashinda dunia, nafsi na Shetani.
\end{stanza}
\begin{stanza}
Tangaza injili iliyo kweli,\\
Umepatanishwa kwa kifo chake;\\
Ukimfuata katika nuru,\\
Damu ya Yesu itakutakasa kweli.
\end{stanza}
\end{hymn}

\begin{hymn}{200}{Tungependa kumwona}{Tungependa kumwona...\#392}
\begin{stanza}
Tungependa kumwona Yesu kwani,\\
Maisha yetu yapita hima,\\
Tumwone ili imani izidi,\\
Na tuwe thabiti hadi mwisho.
\end{stanza}
\begin{stanza}
Tumwone Yesu, msingi wa mwamba,\\
Ambao twasimama juu yake,\\
Usumbufu wa maisha na kifo\\
Hautatutoa tukimwona.
\end{stanza}
\begin{stanza}
Tumwone Yesu, kwa kuwa vingine\\
Tulivyopenda vinafifia.\\
Baraka za dunia zinapita\\
Haidhuru kwetu, twakujia.
\end{stanza}
\begin{stanza}
Tumwone Yesu ili tuzidishwe\\
Furaha, nguvu na utayari.\\
Tumwone akituombea sasa,\\
Ili tusikihofie kifo.
\end{stanza}
\end{hymn}

\begin{hymn}{21}{Bwana tufundishe}{Bwana tufundishe...\#34}
