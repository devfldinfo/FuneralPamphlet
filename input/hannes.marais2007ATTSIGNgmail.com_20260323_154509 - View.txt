\begin{hymn}{2}{Yesu ni ujumbe}{\#10}
\begin{stanza}
Yesu ni ujumbe wetu,\\
Tutamtaja Yesu,\\
Tutamtukuza Yesu,\\
Tutamwona Mwokozi.
\end{stanza}
\begin{chorus}
\setchorusline{Yesu tu, hata milele,}
Yesu tu, hata milele,\\
Tumwimbaye ni Yesu,\\
Mwokozi, Bwana, Mfalme,\\
Mwenye heri ajaye.
\end{chorus}
\begin{stanza}
Yesu ni Mwokozi wetu,\\
Atusafisha dhambi,\\
Atuhesabia haki,\\
Anatuimarisha.
\end{stanza}
\begin{stanza}
Yesu ndiye nguvu yetu,\\
Akaa mioyoni mwetu,\\
Tusihofu, tusisite,\\
Atoa neema, nguvu.
\end{stanza}
\begin{stanza}
Yesu ndiye Bwana wetu,\\
Ni vema kumtii,\\
Ee! tumtumikieni,\\
Tuwe na wateule.
\end{stanza}
\end{hymn}

\begin{hymn}{3}{Yesu alitoka kwa Baba}{\#9}
\begin{stanza}
Yesu alitoka kwa Baba,\\
Na alimdhihirisha,\\
Akatuonyesha ukweli,\\
Tujue kusudi lake.
\end{stanza}
\begin{chorus}
\setchorusline{Na tufuate siku zote,}
Na tufuate siku zote,\\
Katika hatua zake,\\
Tukimtegemea Yesu,\\
Tushinde kwa jina lake.
\end{chorus}
\begin{stanza}
Giza zito linatawala\\
Pande zote duniani,\\
Kama tukifuata Yesu\\
Tutaenenda nuruni.
\end{stanza}
\begin{stanza}
Alitoa maisha yake,\\
Wafungwa waokolewe,\\
Akatuma mitume wake,\\
Waeleze njia yake.
\end{stanza}
\begin{stanza}
Atakaporudi Mwokozi,\\
Atamshinda Shetani,\\
Ndipo giza litatoweka,\\
Mungu atatufariji.
\end{stanza}
\end{hymn}

\begin{hymn}{4}{Mungu wa Betheli wewe}{\#12}
\begin{stanza}
Mungu wa Betheli wewe,\\
Ndiwe Mchungaji,\\
Ulichunga baba zetu,\\
Utuchunge vivyo.
\end{stanza}
\begin{stanza}
Tumefika sasa mbele\\
Ya kiti cha enzi,\\
Mbele zako tunaweka\\
Nadhiri na dua.
\end{stanza}
\begin{stanza}
Uongoze miguu yetu,\\
Tusibabaike,\\
Tupe mahitaji yetu\\
Yote ya kiroho.
\end{stanza}
\begin{stanza}
Utufunike kwa mbawa\\
Zako hadi mwisho,\\
Ndipo tutakapodumu\\
Nyumbani kwako juu.
\end{stanza}
\begin{stanza}
Ee! utubariki hivyo,\\
Twakusihi sana,\\
Wewe tu ni Mungu wetu,\\
Sasa na daima.
\end{stanza}
\end{hymn}

\begin{hymn}{5}{Mwokozi usiniache}{\#344}
\begin{stanza}
Mwokozi usiniache,\\
Nishike mkono wangu,\\
Kivuli cha mbawa zako,\\
Ki juu yangu kuonyesha\\
Mimi ni wako tu.
\end{stanza}
\begin{stanza}
Napenda kuwepo kwako,\\
Uniondolee shaka,\\
Nisikie sauti yako,\\
Ikiniamsha moyoni,\\
Nifikie mede.
\end{stanza}
\begin{stanza}
Ulindaye roho yangu,\\
Uwe karibu kila saa,\\
Vita inikabilipo,\\
Na giza zito la dhambi,\\
Sitafadhaika.
\end{stanza}
\begin{stanza}
Yanifurahisha mno,\\
Kuguswa nawe ee! Bwana,\\
Hata kukuegemea\\
Kifuani kwa upendo,\\
Ee! mimi ni wako.
\end{stanza}
\end{hymn}
