\begin{hymn}{253}{I'm satisfied in Jesus now}{}
\begin{stanza}
I'm satisfied in Jesus now,\\
My restless soul is calm and still;\\
My weary heart has found its home,\\
My joy it is to do His will.\\
Then sing, my soul, in sweetest song,\\
My captive spirit now is free;\\
At His behest I follow on -\\
His, only His, henceforth to be.
\end{stanza}
\begin{chorus}
\setchorusline{Oh! fellowship supremely sweet,}
Oh! fellowship supremely sweet,\\
Oh! matchless love so pure, divine;\\
My soul has found a sure retreat -\\
The lowly Jesus now is mine.
\end{chorus}
\begin{stanza}
His love has overpowered my heart,\\
No longer I in sin repine;\\
Secure from Satan's fiery dart,\\
On Jesus' breast I now recline.\\
I hear His kindly whispered word;\\
With beauty rare His face doth shine;\\
I feel the pressure of His hand,\\
Assuring me that He is mine.
\end{stanza}
\begin{stanza}
I'll follow Him, the sinless One,\\
And all His marvellous way adore,\\
Until I stand complete in Him,\\
His image bear for evermore.\\
When life's short journey here will end,\\
And I at last shall reach the goal,\\
What rapture to behold His face,\\
While the eternal ages roll!
\end{stanza}
\end{hymn}

\begin{hymn}{42}{There is a way}{}
\begin{stanza}
There is a way, a narrow way,\\
That leads to life above -\\
A way of peace and holiness,\\
Of purity and love.
\end{stanza}
\begin{chorus}
\setchorusline{My yearning soul desires to find}
My yearning soul desires to find\\
This hidden path of bliss;\\
Lord, hear my prayer, and lead me in\\
The way of holiness.
\end{chorus}
\begin{stanza}
To find this way, this living way,\\
We must forego our sin:\\
God has declared the pure in heart\\
Alone can enter in.
\end{stanza}
\begin{stanza}
There is a way, a humble way,\\
A way of truth and grace -\\
The highway of God's righteousness,\\
Where self can have no place.
\end{stanza}
\begin{stanza}
There is a way, a perfect way;\\
His chosen walk therein;\\
And they who share His suffering now\\
Eternal glory win.
\end{stanza}
\end{hymn}

\begin{hymn}{21}{If we but knew}{}
\begin{stanza}
If we but knew the cost at which He came,\\
The price whereby the veil was rent in twain,\\
Would we not praise as angels praise His name?\\
If we but knew! If we but knew!
\end{stanza}
\begin{stanza}
If we but knew the sorrow and the loss,\\
The lonely hours, the garden, yea, the cross,\\
Before such love all else would be as dross,\\
If we but knew! If we but knew!
\end{stanza}
\begin{stanza}
If we but knew the joy His heart has planned,\\
The strength and mercy of the outstretched hand,\\
Not long would He rejected waiting stand,\\
If we but knew! If we but knew!
\end{stanza}
\begin{stanza}
If we but knew! O Jesus, Lord of all,\\
Before whom angels bow and nations fall,\\
Lest we resist Thy sweet insistent call,\\
Help us to know, help us to know.
\end{stanza}
\end{hymn}

\begin{hymn}{52}{The king of kings}{}
\begin{stanza}
The King of kings is very near thee,\\
E'en though thy path be dark and lone;\\
How tenderly He waits to cheer thee.\\
If thou wilt make Him now thine own.
\end{stanza}
\begin{chorus}
\setchorusline{The King of kings is gently pleading;}
The King of kings is gently pleading;\\
O do not let Him then depart!\\
His grace and love thy soul is needing;\\
Enthrone Him King within thy heart.
\end{chorus}
\begin{stanza}
Dark night of fears that doth affrighten\\
Shall vanish in the light of day;\\
Thy heavy load of sin shall lighten:\\
He'll wash the stains of guilt away.
\end{stanza}
\begin{stanza}
In changing life, a friend unfailing\\
This King of kings will be to thee;\\
When Jordan's waves are fierce assailing,\\
How near and dear He then will be!
\end{stanza}
\end{hymn}

\begin{hymn}{32}{Die stem van Jesus roep}{\#114}
\begin{stanza}
Die stem van Jesus roep my nou.\\
Graag staan ek op en gaan,\\
By Hom wil ek in vrede woon,\\
Hy bied vergifnis aan.
\end{stanza}
\begin{chorus}
\setchorusline{Ek sal gaan, belas met sonde,}
Ek sal gaan, belas met sonde,\\
Moeg, van skuld bewus,\\
Hoe onwaardig ook, ontvang my\\
Tot beloofde rus.
\end{chorus}
\begin{stanza}
My gees dit ken die vrede nie,\\
Onrustig dwaal ek rond,\\
Maak so my smartlik' laste swaar,\\
Verkort my lewenstond.
\end{stanza}
\begin{stanza}
Genade groot hou Hy my uit\\
Wat troos en red en lei.\\
n Weiveld altyd fris en groen\\
En waterstrome vry.
\end{stanza}
\begin{stanza}
Ek hoor Hy roep herhalend weer\\
In sagte klanke soet:\\
``'n Rus wat ewig rus sal wees,\\
Wag u vermoeide voet.''
\end{stanza}
\end{hymn}

\begin{hymn}{62}{Gee aan die Meester}{\#33}
\begin{stanza}
Gee aan die Meester u beste,\\
Gee van u jeugvolle krag,\\
Werp uself hart en met siel in,\\
Stry vir die waarheid met mag.\\
Jesus stel ons steeds die voorbeeld,\\
Moedig en dapper is Hy,\\
Gee Hom lojaal u gehegtheid,\\
Gee Hom u beste en stry.
\end{stanza}
\begin{chorus}
\setchorusline{Gee aan die Meester u beste,}
Gee aan die Meester u beste,\\
Gee van u jeugvolle krag,\\
Met volle 'rusting gewapen\\
Stry vir die waarheid met mag.
\end{chorus}
\begin{stanza}
Gee aan die Meester u beste,\\
Gee Hom die troon van u hart,\\
Stel Hom die eerste in alles,\\
Wy Hom geheel elke part.\\
Gee en aan u sal gegee word,\\
God het sy Liefste gegee,\\
Dien Hom uit dankbaarheid trou dan,\\
Wil Hom u beste bestee.
\end{stanza}
\begin{stanza}
Gee aan die Meester u beste,\\
Niks minder kan u Hom gee.\\
Hy gee sy lewe as losprys\\
En ook sy heerlikheid mee.\\
Lê neer sy lewe vrywillig\\
U van die sonde te red.\\
Gee Hom u volle aanbidding,\\
Gee Hom selfs al wat u het.
\end{stanza}
\end{hymn}
