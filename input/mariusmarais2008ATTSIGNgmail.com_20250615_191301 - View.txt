\begin{hymn}{23}{Deur die nag van smart}
\hymntitlewithdropcap{23}{Deur die nag van smart...\#334}
\begin{stanza}
Deur die nag van smart en twyfel\\
Gaan die pelgrimsbende voort;\\
Met 'n lied vol bly' verwagting;\\
Hemel hul bestemmingsoord.
\end{stanza}
\begin{stanza}
Helder deur die duister skynend\\
Straal die lig wat leiding gee.\\
Broederhand in hand van broeder\\
Onbevrees die weg betree.
\end{stanza}
\begin{stanza}
Diep bewus van God se bysyn:\\
Dis soos dou wat op ons daal,\\
Dit verdryf die vrees so somber.\\
Met 'n glans ons pad bestraal.
\end{stanza}
\begin{stanza}
Een die doelwit van ons reise;\\
Een die onvermoeid' geloof;\\
Een die oog wat blik na vore:\\
Een die hoop. slegs een die Hoof.
\end{stanza}
\begin{stanza}
Een die lied wat duisend tonge\\
Hef soos uit die hart van een;\\
Een die worst'ling teen die vyand;\\
Een die weg deur God geseën.
\end{stanza}
\begin{stanza}
Een die blydskap van verheuging\\
Wat aan hemelkus ons wag;\\
Een die Vader wat in liefde\\
Heers tot laaste nageslag.
\end{stanza}
\end{hymn}

\begin{hymn}{20}{Daar is 'n Heiland}
\hymntitlewithdropcap{20}{Daar is 'n Heiland...\#156}
\begin{stanza}
Daar is 'n Heiland, sy liefde duur ewig.\\
Hy wag in genade op u en op my,\\
Hy wil u omhels en u kroon met sy vrede.\\
Kom dan tot die Heiland, sy lief' is so vry.
\end{stanza}
\begin{chorus}
\setchorusline{Ook u roep die Heiland;}
Ook u roep die Heiland;\\
Ook u roep die Heiland;\\
Ook u roep die Heiland;\\
Hy roep u reeds nou.
\end{chorus}
\begin{stanza}
Smaak van sy goedheid terwyl Hy dit aanbied,\\
Die vreugd' van sy koninkryk u onbekend.\\
Die Herder self lei u, Hy help en beskerm u.\\
Wil dan tog betyds nog u hart na Hom wend.
\end{stanza}
\begin{stanza}
Hy gaan verby en miskien reeds vir altyd\\
Waardeur die geleentheid nou nog so naby.\\
Neem wat Hy u aanbied, nou is daar die tyd nog.\\
O neem dan die stap tot die ingang so vry.
\end{stanza}
\end{hymn}

\begin{hymn}{27}{Die Heiland roep dringend}
\hymntitlewithdropcap{27}{Die Heiland roep dringend...\#122}
\begin{stanza}
Die Heiland roep dringend, neig Hom tog u oor.\\
Die hemel wag angstig of u sy stem hoor;\\
Bedroef nie sy Gees nie, doen Hom tog geen smart;\\
Verlaat Hy u, bly daar geen hoop in u hart.
\end{stanza}
\begin{stanza}
Die deur is nou wyd oop; sy goedheid is groot.\\
Wat staan u te wag dan miskien kom die dood;\\
Die bose wil hinder: ``Wag nog maar 'n dag.''\\
Hoe jammer te sterwe in pikdonker nag.
\end{stanza}
\begin{stanza}
In duisternis voortgaan, sy lig te ontsê:\\
Onwillig te dien Hom, ontken u die reg;\\
Versmaad u die Heiland, hoe vreeslik u lot;\\
Verwerp u die Heiland dan het u geen God.
\end{stanza}
\end{hymn}

\begin{hymn}{24}{Deur eer en oneer}
\hymntitlewithdropcap{24}{Deur eer en oneer...\#298}
\begin{stanza}
Deur eer en oneer trou bewaar.\\
Red U ons. Heer, uit elk' gevaar.\\
U is ons staf, ons skild, ons swaard;\\
Ons volg U na.
\end{stanza}
\begin{stanza}
Omring deur magte aan elk' kant\\
Verlos U ons met sterke hand;\\
Bevry van elke aardse band;\\
Volg ons U na.
\end{stanza}
\begin{stanza}
O Meester, toon U ons die weg;\\
Teen al wat kwaad is leer ons veg,\\
Dan op u weg getrou, opreg;\\
Volg ons U na.
\end{stanza}
\begin{stanza}
U het vir ons die weg gebaan,\\
Waardeur ons nou tot God mag gaan.\\
Gee moed en help ons vas te staan;\\
Ons volg U na.
\end{stanza}
\begin{stanza}
Wie het ons buiten U, O Heer,\\
In hemel of op aarde meer?\\
Bestraal ons met u liefde teer;\\
Ons volg U na.
\end{stanza}
\end{hymn}

\begin{hymn}{22}{Daar kom donker dae}
\hymntitlewithdropcap{22}{Daar kom donker dae...\#261}
\begin{stanza}
Daar kom donker dae in die lewe hier,\\
Wanneer ek alles nie verstaan,\\
Maar ek weet u oog hou die wag omhoog:\\
Skenk gena' om steeds aan te gaan.
\end{stanza}
\begin{chorus}
\setchorusline{Help my sê: ``Nie my wil eskied.''}
Help my sê: ``Nie my wil eskied.''\\
Help my doen al wat U gebied.\\
Gee dan krag aan my en genade vry\\
Om te sê: ``Heer, u wil geskied.''
\end{chorus}
\begin{stanza}
Heer, ek's nie bekwaam om my weg te rig;\\
Ook geen stap kan ek voorwaarts gaan.\\
Wees my steeds naby; laat u hand my lei,\\
Of ek dwaal van die lewensbaan.
\end{stanza}
\begin{stanza}
Help my, Heer, te wees soos 'n plantjie teer\\
Wat kan buig in u god'lik' hand.\\
Heer kom woon in my; hou my aan u sy\\
Ek ontbreek wysheid en verstand.
\end{stanza}
\begin{stanza}
Help my, Heer, te sien soos ek dan sal sien\\
Wanneer tyd hier bestaan nie meer.\\
Ek sal antwoord: ``Ja,'' wat U ook al vra\\
En ek kniel dankbaar voor U neer.
\end{stanza}
\end{hymn}
