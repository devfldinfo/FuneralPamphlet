\begin{hymn}{200}{Ons sien graag Jesus}{\#392}
\begin{stanza}
Ons sien graag Jesus, skaduwees word langer\\
Oor die klein landskap van ons lewensduur;\\
Ons sien graag Jesus, ons geloof versterk Hy\\
Ook gee Hy krag in laaste lewensuur.
\end{stanza}
\begin{stanza}
Ons sien graag Jesus, Hy, ons Rots en Vesting\\
Waarop ons voet geplaas is deur gena'.\\
Nie dood of lewe met hul verontrusting\\
Kan ons verhinder want ons volg Hom na.
\end{stanza}
\begin{stanza}
Ons sien graag Jesus – ander ligte kwyn nou,\\
Wat ons vertroosting was in tyd verby;\\
Ons eie kragte kwyn ook, word al minder;\\
Ons treur nie, want ons gaan met U te bly.
\end{stanza}
\begin{stanza}
Ons sien graag Jesus – Ons het Hom so nodig;\\
Krag, vreug', gewilligheid kom met die sig;\\
Ons sien Hom sterwend, opgestaan en pleitend;\\
Vaarwel nou nag, ons groet 'n blye dag.
\end{stanza}
\end{hymn}

\begin{hymn}{54}{So kind a shepherd}{}
\begin{stanza}
Was there e'er so kind a Shepherd,\\
One so gentle yet so great,\\
As the Saviour, who would have us\\
Come and gather round His feet?
\end{stanza}
\begin{stanza}
There's a wideness in God's mercy,\\
Like the wideness of the sea;\\
There's a kindness in His justice,\\
Which is more than liberty.
\end{stanza}
\begin{stanza}
There is no place where earth's sorrows\\
Are more felt than up in heaven;\\
There is no place where earth's failings\\
Have such kindly judgment given.
\end{stanza}
\begin{stanza}
For the love of God is broader\\
Than the measure of man's mind,\\
And the heart of the Eternal\\
Is most wonderfully kind.
\end{stanza}
\end{hymn}

\begin{hymn}{20}{Net Christus Kan Tevrede Stel}{\#109}
\begin{stanza}
In U, o Heer het ek gevind,\\
Gevind in U alleen,\\
Die vree', geluk waarna ek soek;\\
Die vreugde ongekend.
\end{stanza}
\begin{chorus}
\setchorusline{Net Christus kan my tevrede stel,}
Net Christus kan my tevrede stel,\\
Geen ander naam vir my.\\
Daar's liefde, lewe, sielsgenot,\\
In Jesus te verkry!
\end{chorus}
\begin{stanza}
Na rus en vreugd' het ek verlang,\\
Maar nie na U, o Heer.\\
Terwyl ek U verby wou gaan\\
Het liefde my gekeer.
\end{stanza}
\begin{stanza}
Leë vreugde het ek toe probeer;\\
Die bronne stel teleur.\\
Terwyl ek nog daarna wou reik,\\
Word dors in my al meer.
\end{stanza}
\begin{stanza}
Verlore vreugd' het ek beween,\\
Maar nooit oor U getreur\\
Totdat ek deur genade sien,\\
U lieflikheid, o Heer.
\end{stanza}
\end{hymn}

\begin{hymn}{200}{Lord, my heart's deep need}{}
\begin{stanza}
Lord, my heart's deep need Thou knowest -\\
Contrite at Thy feet I pray,\\
Watching, waiting in Thy presence,\\
Listening, willing to obey.\\
All my soul for Thee is yearning,\\
Fainting for Thy courts, O Lord;\\
Truest source of satisfaction,\\
Feed me on the living word.
\end{stanza}
\begin{chorus}
\setchorusline{All my soul for Thee is yearning,}
All my soul for Thee is yearning,\\
To be filled with Thee alone,\\
Sanctified, approved, accepted,\\
One whom Thou canst seal and own.
\end{chorus}
\begin{stanza}
Lord my heart's deep need Thou knowest -\\
Draw me, I will follow Thee,\\
Free from sin's enthralling bondage,\\
In Thy light and liberty.\\
All my weakness, Lord, Thou seest;\\
In temptation be my tower;\\
Make me more and more like Jesus\\
Daily by Thy Spirit's power.
\end{stanza}
\begin{stanza}
Lord, my heart's deep need Thou knowest -\\
Do not turn away my prayer;\\
Thou hast walked the path before me;\\
Thou canst sympathize and care.\\
O remember me in mercy,\\
And impart the needed grace\\
To enable me to follow\\
Till I see Thee face to face.
\end{stanza}
\end{hymn}
