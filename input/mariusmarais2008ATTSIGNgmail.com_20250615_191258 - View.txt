\begin{hymn}{23}{God's word is so pure}
\hymntitlewithdropcap{23}{God's word is so pure...\#29}
\begin{stanza}

Ngithanda iz'fundo zalo,\\
Lingikhanyisela indlela yami,\\
Luze luphele uhambo.
\end{stanza}

\setchorusline{Kudlula konke okwasemhlabeni,}

Kukuhle kunegolide,\\
Ukuhlala ngasez'nyaweni zaKhe,\\
Ngizwe Ekhuluma kimi.
\end{chorus}
\begin{stanza}
Imnandi, inhle indlela yeNkosi\\
La ngibuswa nguMsindisi;\\
Izwi Alishoyo liyaphilisa,\\
Lingivuselela njalo.
\end{stanza}
\begin{stanza}
Ngiyezwa, ngilalele, ngikhululwe\\
Ukulandela uJesu;\\
Ngigcina izwi laKhe enhliz'yweni,\\
NgiMdumise ngenjabulo.
\end{stanza}
\begin{stanza}
Ngiqinise ezimisweni zaKho,\\
Ngithande umthetho waKho;\\

Ukuzindla ngawo njalo.
\end{stanza}
\end{hymn}

\begin{hymn}{44}{I was a wandering sheep}
\hymntitlewithdropcap{44}{I was a wandering sheep}
\begin{stanza}
Ngazula njengemvu\\
Kude nesibaya;\\
NgingaMthandi uMelusi,\\
Ngala ukukhuzwa;\\
Ngangingalaleli,\\
Ngingathand' ikhaya,\\
Ngingathand' izwi leNkosi,\\
Ngathand' ukuzula.
\end{stanza}
\begin{stanza}
Ngafunwa nguYena,\\
UMelusi wami,\\
Ngafunwa ezigodini\\
Nasezintabeni:\\
Ngafunyanwa ngifa\\
Yindlala, ngingedwa;\\
Ngaboshwa luthando lwaKhe,\\
Ngaze ngasindiswa.
\end{stanza}
\begin{stanza}
Ngathandwa nguJesu,\\
UMelusi wami;\\
Wangihlanza egazini,\\
Wangiphulukisa;\\
Ngafunwa, ngatholwa\\
Ngisekude naYe,\\
Ngangeniswa es'bayeni\\
NguYe onguMlondi.
\end{stanza}
\begin{stanza}
Ngazula njengemvu,\\
Ngala ukuphathwa;\\
Konke sekukusha manje,\\
Ngithanda is'baya.\\
Kade nginenkani\\
Ngisazulazula;\\
Manje senginoMelusi\\
NgiMthobela njalo.
\end{stanza}
\end{hymn}

\begin{hymn}{102}{How clear the call of Jesus}
\hymntitlewithdropcap{102}{How clear the call of Jesus...\#147}
\begin{stanza}
UJesu usibiza\\
Ngezwi elimnandi:\\
``Woza, okhatheleyo,\\
Phumuza iz'nyawo;\\
Ngizokuthethelela,\\
Ngikuhole kahle,\\
Ngasemanzin' empilo,\\
Ube nokuthula.''
\end{stanza}
\begin{stanza}
UngaMali uJesu;\\
Uyithemba lakho;\\
Wokuma phambi kwaKhe\\
Ngosuk' oluzayo,\\
Lapho izizwe zonke\\
Zizokwahlulelwa,\\
Bonke abaMdelela\\
Behlelwe luvalo.
\end{stanza}
\begin{stanza}
UJesu usabiza\\
Ngobubele baKhe,\\
UngaMnqabi uMoya\\
UngaMfulatheli;\\
Izinsuku zidlula\\
Njengenkungu ngokusa;\\
Wosala yini phandle\\
Kwesibaya saKhe?
\end{stanza}
\end{hymn}
