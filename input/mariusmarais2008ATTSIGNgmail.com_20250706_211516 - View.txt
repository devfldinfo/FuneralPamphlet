\begin{hymn}{32}{Ons Verlosser}{\#208}
\begin{stanza}
Net voor sy tere afskeidsgroet,\\
Het Jesus ons sy Gees\\
As Trooster en as Gids beloof\\
By ons te wees.
\end{stanza}
\begin{stanza}
'n Soete invloed bring Hy mee,\\
Ons Vriend is Hy gewis.\\
Hy soek nou na 'n ned'rig hart\\
Om in te rus.
\end{stanza}
\begin{stanza}
En syne is die sagte stem\\
Wat op ons harte daal,\\
Gedagtes tem en vrees verdryf\\
Met hemelstaal.
\end{stanza}
\begin{stanza}
En elke deug wat ons besit\\
Selfs as oorwinnings kom.\\
En elk' gedagte goddelik,\\
Is slegs van Hom.
\end{stanza}
\begin{stanza}
O, Gees van reinheid en gena',\\
Sien op ons swakheid neer.\\
Kom maak ons hart u woonplek nou;\\
Meer waardig, Heer.
\end{stanza}
\end{hymn}
