\begin{hymn}{234}{Bwana giza la jioni}
\hymntitlewithdropcap{234}{Bwana giza la jioni}
\begin{stanza}
Bwana giza la jioni lafika,\\
Laiti siku iwe ndefu zaidi,\\
Mavuno yamekwisha iva sana,\\
Ndiyo kwanza naanza kuvuna,\\
Bali ukiniita nije kwako,\\
Kabla kazi hii haijakwisha\\
Au ikiwa tumeshavuna yote,\\
Nisaidie nikujibu, ``Nipo Bwana.''
\end{stanza}
\begin{chorus}
\setchorusline{Ukirudi usiku wa manane,}
Ukirudi usiku wa manane,\\
Hata ukungu wa \al\fajiri,\\
Kusimamia utawala wako,\\
Bwana wangu, mpendwa, unikumbuke
\end{chorus}
\begin{stanza}
Ee! wengine walikutumikia,\\
Kwa mito walitumia mawe,\\
Wengine meli zao zilivunjika,\\
Hawakuhofu shida yo yote,\\
Nipake mafuta nifanye kazi,\\
Pamoja nao mashujaa hawa,\\
Nisaidie nivune mashambani\\
Nilete mavuno ghalani mwako juu.
\end{stanza}
\begin{stanza}
Niko tayari kwenda kuhubiri,\\
Vema zaidi nitumike kwako,\\
Natamani nafasi japo ndogo,\\
Napenda kutimiza wajibu,\\
Nitayashika mausia yako,\\
Hadi mwisho wa maisha yangu,\\
Nikilala kifoni, urudipo,\\
Ee! Bwana niamshe niwe kwako daima.
\end{stanza}
\end{hymn}

\begin{hymn}{49}{When first we heard the message}
\hymntitlewithdropcap{49}{When first we heard the message...\#361}
\begin{stanza}
Tshimologông ya 'êtô\\
Re bolotse mmôgô\\
Go kgatlhantsha monyadi.\\
Ra nyatsa mathata\\
Jaaka makgarebe bao.\\
Mo bošheng re 'fa kae?\\
A re basilo kampo\\
Gare ga matlhale?
\end{stanza}
\begin{chorus}
\setchorusline{``Tsogang, bakaanyang 'pône.''}
``Tsogang, bakaanyang 'pône.''\\
Ka bosigogare\\
Re tl\aa\ utlwa: ``M\ong\ ke yo!\\
Tsw\ang\, lo Mo kgatlhantshê.''
\end{chorus}
\begin{stanza}
Têkô di ôkêtsêga,\\
Lefifi le lone.\\
A re tshwênngwa ke tshabô?\\
Nnyaa, a re itshôkêng!\\
Bangwe ba thulametse\\
B'amogwa tumelô.\\
A 'rato le la rona\\
Le tlaa tsidifala?
\end{stanza}
\begin{stanza}
A re nneng boikanngô\\
Re tlhokomologeng\\
Ba 'boêlang morago.\\
Tsholofelô ke e:\\
Disang, re Mo êmêlê\\
Sigông jwa bodutu.\\
Pône di tle di tuke\\
Mogang go kuiwang:
\end{stanza}
\end{hymn}
