\begin{hymn}{10}{Yesu alikufa}
\hymntitlewithdropcap{10}{Yesu alikufa}
\begin{stanza}
Yesu alikufa kwa wenye dhambi,\\
Acha dhambi zako umwone Mungu;\\
Patana na Yesu, utakuwa hai,\\
Kristo ndani ni taraja la utukufu.
\end{stanza}
\begin{chorus}
\setchorusline{Umkiri na mwamini, Yesu, Mwokozi,}
Umkiri na mwamini, Yesu, Mwokozi,\\
Roho asitoke kwako daima.
\end{chorus}
\begin{stanza}
Akuita wewe, mwenye mizigo,\\
Atakupokea akupumzishe;\\
Muda wa rehema, ziache dhambi,\\
Mungu atakusamehe akupe raha.
\end{stanza}
\begin{stanza}
Ukisamehewa kwa kifo chake,\\
Utaokolewa kwa uhai wake;\\
Akaavyo mwako usiogope,\\
Utashinda dunia, nafsi na Shetani.
\end{stanza}
\begin{stanza}
Tangaza injili iliyo kweli,\\
Umepatanishwa kwa kifo chake;\\
Ukimfuata katika nuru,\\
Damu ya Yesu itakutakasa kweli.
\end{stanza}
\end{hymn}

\begin{hymn}{100}{Ee! Mungu unisikie}
\hymntitlewithdropcap{100}{Ee! Mungu unisikie}
\begin{stanza}
Ee! Mungu unisikie\\
Kilio, maombi,\\
Toka mwisho wa dunia\\
Ninakulilia.
\end{stanza}
\begin{stanza}
Ninapozimia moyo,\\
Na kufadhaika,\\
Niongoze juu ya mwamba,\\
Ulio juu sana.
\end{stanza}
\begin{stanza}
Ulikuwa kimbilio,\\
Ngome yenye nguvu,\\
Na hata kinga, adui\\
Na wasinipate.
\end{stanza}
\begin{stanza}
Hemani kwako nitakaa\\
Nitakimbilia,\\
Sitara ya mbawa zako,\\
Kukaa hata daima.
\end{stanza}
\begin{stanza}
Ee! Mungu nadhiri zangu\\
Umezisikia,\\
Umenipa urithi na\\
Wakuheshimuo.
\end{stanza}
\end{hymn}

\begin{hymn}{121}{Kila asubuhi}
\hymntitlewithdropcap{121}{Kila asubuhi...\#187}
\begin{stanza}
Kila asubuhi,\\
Mwombe Baba Mungu,\\
Tafuta uso wake na\\
Shiriki upendo.
\end{stanza}
\begin{stanza}
Fungua Biblia,\\
Soma fungu humo,\\
Litakase nia yako,\\
Na kukuhifadhi.
\end{stanza}
\begin{stanza}
Shinda siku kutwa\\
Naye Baba Mungu,\\
Atakuwepo karibu,\\
Po pote ulipo.
\end{stanza}
\begin{stanza}
Uongee naye,\\
Kwenye moyo wako,\\
Kiri wema wake wote,\\
Utoe shukrani.
\end{stanza}
\begin{stanza}
Mwishoni mwa siku,\\
Kiri dhambi zako,\\
Amini damu ya Yesu,\\
Na usamehewe.
\end{stanza}
\end{hymn}

\begin{hymn}{142}{Kukubaliwa nawe Mungu}
\hymntitlewithdropcap{142}{Kukubaliwa nawe Mungu...\#299}
\begin{stanza}
Kukubaliwa nawe Mungu tu,\\
Wazo hilo latutia moyo,\\
Tujitoe kwa furaha tuwe\\
Dhabihu iliyo hai kwako.
\end{stanza}
\begin{stanza}
Yesu alikubaliwa naye,\\
Alivumilia, hakushindwa,\\
Akamaliza safari yake,\\
Akafufuka kwenda enzini.
\end{stanza}
\begin{stanza}
Kukubaliwa na Mungu nami,\\
Nitakuwa na mfano wake,\\
Japo napata mateso kama\\
Watu wa kale nifuate tu.
\end{stanza}
\begin{stanza}
Kukubaliwa ni baraka kuu,\\
Heshima ya watu yapita tu,\\
Bwana shika moyo wangu niwe\\
Kama Yesu, mwaminifu kweli.
\end{stanza}
\end{hymn}

\begin{hymn}{150}{Roho yangu isijazwe}
\hymntitlewithdropcap{150}{Roho yangu isijazwe...\#340}
\begin{stanza}
Roho yangu isijazwe huzuni,\\
Njia yangu ikiwa ya shida,\\
Kwa kuwa Bwana hulinda daima,\\
Mkono wake wanipa nguvu.
\end{stanza}
\begin{chorus}
\setchorusline{Leo Bwana, ninakuamini,}
Leo Bwana, ninakuamini,\\
Nikuamini daima.
\end{chorus}
\begin{stanza}
Unifunge katika madhabahu,\\
Nami kwa hiari najitoa,\\
Unijalie Bwana nisisite,\\
Bali nifanye vyema daima.
\end{stanza}
\begin{stanza}
Mataraja yangu yasidhoofu,\\
Bali yaimarishwe na Yesu,\\
Utanifufua pamoja nao\\
Watu wako utakaporudi.
\end{stanza}
\end{hymn}

\begin{hymn}{42}{Kuoshwa kila waa}
\hymntitlewithdropcap{42}{Kuoshwa kila waa...\#24}
\begin{stanza}
Kuoshwa kila waa, kufidiwa dhambi,\\
Hiki ni kipawa chako ulichoahidi,\\
Nakiri makosa, udhaifu wangu,\\
Nafika jinsi nilivyo mbele zako Mungu.
\end{stanza}
\begin{stanza}
Sikufichi kitu, dhambi na makosa,\\
Wala kiburi moyoni ninachochukia.\\
Jinsi nilivyo tu, mbele zako Bwana,\\
Mnyonge tu mwenye hamu ya uzima wako.
\end{stanza}
\begin{stanza}
Nilipo nuruni, najiona moyo,\\
Nimekosa kupokea kipawa cha Mungu,\\
Ee! Bwana neema, hekima na nguvu\\
Zako zitanihifadhi, naamini hivyo.
\end{stanza}
\begin{stanza}
Juu ya madhabahu naweka maisha,\\
Napenda uwe karibu, kuwa Mtawala.\\
Mungu uondoe mawazo yangu tu,\\
Ili niliamini tu jina lako kuu.
\end{stanza}
\end{hymn}
