\begin{hymn}{210}{Bwana usinyamaze}
\hymntitlewithdropcap{210}{Bwana usinyamaze...\#227}
\begin{stanza}
Ee! Bwana usinyamaze,\\
Bali kwa rehema nena,\\
Nchi ingojavyo mvua,\\
Ndivyo nikusubirivyo.
\end{stanza}
\begin{chorus}
\setchorusline{Nakutegemea Bwana,}
Nakutegemea Bwana,\\
Naona kwako taraja,\\
Kwani umenifariji,\\
Nyakati za taabu nyingi.
\end{chorus}
\begin{stanza}
Bwana ukinena kuonya\\
Nitaona heri kwangu,\\
Maana aniadhibuye\\
Ndiye anayenipenda.
\end{stanza}
\begin{stanza}
Bwana unioshe sasa\\
Kwa neno lako la tunu,\\
Ee! uniponye nafsini,\\
Unirudishie nguvu.
\end{stanza}
\begin{stanza}
Bwana nifanye mateka,\\
Ni radhi kuwa mfungwa,\\
Nitende mapenzi yako,\\
Nitawaliwe na wewe.
\end{stanza}
\end{hymn}

\begin{hymn}{21}{Bwana tufundishe}
\hymntitlewithdropcap{21}{Bwana tufundishe...\#34}

Ya kuhesabu siku,\\

Kukusifu daima.
\end{stanza}
\begin{chorus}
\setchorusline{Bwana tuwe na hekima,}

\end{chorus}
\begin{stanza}

Ili tutende vema.
\end{stanza}
\begin{stanza}

\end{stanza}
\begin{stanza}

Kikomo cha safari.
\end{stanza}
\end{hymn}

\begin{hymn}{215}{Usilegeze ushujaa}
\hymntitlewithdropcap{215}{Usilegeze ushujaa...\#343}
\begin{stanza}
Usilegeze ushujaa kutenda haki,\\
Maana Bwana atakuongoza salama,\\
Adui watakujaribu, watashindwa,\\
Washikamanao na Mungu watas'tawi.
\end{stanza}
\begin{chorus}
\setchorusline{Usilegeze ushujaa,}
Usilegeze ushujaa,\\
Uwe thabiti,\\
``Bwana unichunge niwe\\
Dhabihu kwako.''
\end{chorus}
\begin{stanza}
Usichoke moyoni japo njia ndefu,\\
Panda kwa tumaini nawe utavuna,\\
Na Mungu atatimiza ahadi zake,\\
Basi na umpendeze kwa imani kuu.
\end{stanza}
\begin{stanza}
Usitazame nyuma kwenye vita kali,\\
Kaza macho mbele katika safari hii,\\
Japo mwisho waonekana kuwa mbali,\\
Dhiki zetu nyepesi zadumu kitambo.
\end{stanza}
\begin{stanza}
Usiwe mvivu kwani kazi ni kubwa,\\
Sikia mwito wa Mungu kwa kazi hii,\\
Zipo nafasi kwa wote za kutumika,\\
Mavuno yahitaji bidii zetu zote.
\end{stanza}
\end{hymn}
