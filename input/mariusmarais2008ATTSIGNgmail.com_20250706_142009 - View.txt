\begin{hymn}{72}{Why should I walk?}{}
\begin{stanza}
Why should I walk in paths of night\\
Which give no peace nor pure delight?\\
My way is wrong, God's way is right:\\
His way is seen in Jesus.
\end{stanza}
\begin{chorus}
\setchorusline{The truth of God my heart has won;}
The truth of God my heart has won;\\
He'll finish what He has begun;\\
In earth or heaven there is none\\
Who can compare with Jesus.
\end{chorus}
\begin{stanza}
Man's wisdom leads into a maze,\\
And error grows in bypath ways;\\
But hearts are filled with joy and praise,\\
Who see the truth in Jesus.
\end{stanza}
\begin{stanza}
My human power cannot avail;\\
In war with sin it can but fail;\\
By this alone I can prevail:\\
Receiving life in Jesus.
\end{stanza}
\begin{stanza}
Though sin assail me like a flood,\\
I'll plead His name and precious blood;\\
The pure and spotless Son of God\\
And sinners' friend is Jesus.
\end{stanza}
\end{hymn}

\begin{hymn}{52}{Ek is tevrede in my Heer}{\#253}
\begin{stanza}
Ek is tevrede in my Heer;\\
My rust'loos hart is kalm en stil,\\
My siel, so moeg, vind rus in Hom.\\
My vreugde is net in sy wil;\\
My siel sing nou 'n soete lof;\\
My gees is geen gevang'ne meer:\\
Terwyl ek hier sy Naam bely,\\
Volg ek Hom as my Koning, Heer.
\end{stanza}
\begin{chorus}
\setchorusline{Gemeenskap o hoe god'lik rein}
Gemeenskap o hoe god'lik rein\\
Onmeetlik' liefde, hemelsoet.\\
My siel dit ken 'n skuilplek daar,\\
Dis aan my dierbaar' Meesters' voet.
\end{chorus}
\begin{stanza}
Sy liefde het my hart oorwin;\\
Leef in die sonde voort nie meer;\\
Maar veilig van die Satanslis\\
Lê ek aan Jesus' bors my neer.\\
Ek hoor sy liefdevolle stem;\\
Sy aangesig skyn wonderskoon,\\
Terwyl sy tere sagte stem\\
My lewe met genot bekroon.
\end{stanza}
\begin{stanza}
Ek volg Hom na; Gods' sond'loos' Lam;\\
Op weë rein en wonderbaar,\\
Totdat ek staan volmaak voor Hom,\\
Sy beeld in my word openbaar.\\
Breek dan die silwerkoord vir my,\\
Dat ek geen dae hier meer tel,\\
Gaan ek die ewig' vreugde in\\
Waar eeu na eeu soet heen sal snel.
\end{stanza}
\end{hymn}

\begin{hymn}{52}{Bwana napenda kukujia}{\#161}
\begin{stanza}
Bwana napenda kukujia,\\
Umeningojea kwa subira,\\
Upendo wako umenishinda,\\
Hutadharau unyonge wangu.
\end{stanza}
\begin{chorus}
\setchorusline{Bwana nitwae niwe wako,}
Bwana nitwae niwe wako,\\
Nimefidiwa kwa damu yako,\\
Nitawashinda adui wote,\\
Kwa uwezo wako na upendo.
\end{chorus}
\begin{stanza}
Kuona ufalme wa Mungu,\\
Lazima niwe kama mtoto,\\
Naacha yote yanipingayo,\\
Na kujiweka chini yako.
\end{stanza}
\begin{stanza}
Ninadharau ulimwengu huu,\\
Hazina, fahari zilizomo,\\
Hizi hazina maana yo yote,\\
Naziona kama hasara.
\end{stanza}
\begin{stanza}
Nilikuwa nikidanganywa,\\
Nikitangatanga gizani tu,\\
Sasa napenda njia yako tu,\\
Nisaidie nikufuate.
\end{stanza}
\end{hymn}

\begin{hymn}{52}{The king of kings}{}
\begin{stanza}
The King of kings is very near thee,\\
E'en though thy path be dark and lone;\\
How tenderly He waits to cheer thee.\\
If thou wilt make Him now thine own.
\end{stanza}
\begin{chorus}
\setchorusline{The King of kings is gently pleading;}
The King of kings is gently pleading;\\
O do not let Him then depart!\\
His grace and love thy soul is needing;\\
Enthrone Him King within thy heart.
\end{chorus}
\begin{stanza}
Dark night of fears that doth affrighten\\
Shall vanish in the light of day;\\
Thy heavy load of sin shall lighten:\\
He'll wash the stains of guilt away.
\end{stanza}
\begin{stanza}
In changing life, a friend unfailing\\
This King of kings will be to thee;\\
When Jordan's waves are fierce assailing,\\
How near and dear He then will be!
\end{stanza}
\end{hymn}

\begin{hymn}{73}{God verlang nog immer}{\#35}
\begin{stanza}
God verlang nog immer om die mens te wen,\\
Van sy sondig' weë om Gods' weg te ken.\\
Wat die Vader dan van ons op aard' begeer,\\
Is dat Jesus in ons moet herlewe weer.
\end{stanza}
\begin{chorus}
\setchorusline{Redding kan maar net deur Jesus kom;}
Redding kan maar net deur Jesus kom;\\
Ook oorwinning kry ons net deur Hom;\\
Hom as Heer, bely met woord en daad,\\
Is wat God bepaal het in sy ewig' raad.
\end{chorus}
\begin{stanza}
Let ons hier as voorbeeld op 'n medemens?\\
Of wil u deur goeie werke doen Gods' wens?\\
God het ons dit lank reeds deur sy woord laat weet:\\
Ons geregtigheid is net 'n onrein kleed.
\end{stanza}
\begin{stanza}
Wil u eerlik handel met u siel voor God?\\
Maak dan Jesus Meester van geheel u lot.\\
Vrees nooit Hom te dien of hier sy Naam bely,\\
Dan gee Hy u krag en bly steeds aan u sy.
\end{stanza}
\end{hymn}

\begin{hymn}{98}{Herder van Israel}{\#310}
\begin{stanza}
Herder van Israel, U hou steeds wag\\
Oor heel u kudde, bedags en ook snags.\\
En as gevaar kom u arms om hul slaan,\\
Lei U hul voort, U as Herder vooraan.
\end{stanza}
\begin{chorus}
\setchorusline{Herder van Israel, liefdevol' Heer,}
Herder van Israel, liefdevol' Heer,\\
Sien tog ontfermend op u kudde neer.\\
U ken hul pad deur die diepste vallei,\\
Biddend vir hul word hul sag voortgelei.
\end{chorus}
\begin{stanza}
Herder van Israel, u arm bewaar,\\
Beskut u kudde teen dreigend' gevaar.\\
U sal u lammers versamel wat val,\\
Dig aan u boesem omhels U almal.
\end{stanza}
\begin{stanza}
Herder van Israel, kom spoedig weer!\\
Om te verlos u ``klein kudde'' so teer.\\
U te aanskou, wat begeer hul ooit meer,\\
Herder van Israel, Jesus hul Heer.
\end{stanza}
\end{hymn}
