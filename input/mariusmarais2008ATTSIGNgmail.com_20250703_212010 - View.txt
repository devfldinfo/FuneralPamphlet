\begin{hymn}{314}{O give me rest}{}
\begin{stanza}
My Saviour, Thou hast offered rest;\\
O give it, then, to me:\\
The rest of ceasing from myself,\\
To find my all in Thee.
\end{stanza}
\begin{stanza}
This cruel self, oh, how it strives\\
And works within my breast,\\
To come between Thee and my soul,\\
And keep me back from rest!
\end{stanza}
\begin{stanza}
How many subtle forms it takes\\
Of seeming verity,\\
As if it were not safe to rest\\
And venture all on Thee.
\end{stanza}
\begin{stanza}
O Lord, I seek a holy rest,\\
A victory over sin;\\
I seek that Thou alone shouldst reign\\
O'er all, without, within.
\end{stanza}
\begin{stanza}
In Thy strong hand I lay me down:\\
So shall the work be done;\\
For who can work so wondrously\\
As Thou, Almighty One?
\end{stanza}
\end{hymn}

\begin{hymn}{32}{Die stem van Jesus roep}{\#114}
\begin{stanza}
Die stem van Jesus roep my nou.\\
Graag staan ek op en gaan,\\
By Hom wil ek in vrede woon,\\
Hy bied vergifnis aan.
\end{stanza}
\begin{chorus}
\setchorusline{Ek sal gaan, belas met sonde,}
Ek sal gaan, belas met sonde,\\
Moeg, van skuld bewus,\\
Hoe onwaardig ook, ontvang my\\
Tot beloofde rus.
\end{chorus}
\begin{stanza}
My gees dit ken die vrede nie,\\
Onrustig dwaal ek rond,\\
Maak so my smartlik' laste swaar,\\
Verkort my lewenstond.
\end{stanza}
\begin{stanza}
Genade groot hou Hy my uit\\
Wat troos en red en lei.\\
n Weiveld altyd fris en groen\\
En waterstrome vry.
\end{stanza}
\begin{stanza}
Ek hoor Hy roep herhalend weer\\
In sagte klanke soet:\\
``'n Rus wat ewig rus sal wees,\\
Wag u vermoeide voet.''
\end{stanza}
\end{hymn}

\end{chorus}
\begin{stanza}

Yena Othembekile;\\

\end{stanza}
\end{hymn}
