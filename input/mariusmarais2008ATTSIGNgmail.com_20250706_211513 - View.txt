\begin{hymn}{5}{Kwakungenxa yam?}{\#8}
\begin{stanza}
Kwakungenxa yam ndodwa\\
UMsindis'washiy'itrone,\\
Ubukhazi bamazulu\\
Weza kundifela apha?
\end{stanza}
\begin{chorus}
\setchorusline{Ewe weza ngenxa yam,}
Ewe weza ngenxa yam,\\
Ngothando olukaThixo,\\
Olukhulu, olwenene\\
Wandifela Yena, Yesu.
\end{chorus}
\begin{stanza}
Ngenxa yam zamculela,\\
Iingelosi eJudeya\\
Ngobusuku obungcwele.\\
Walungisela mna Thixo?
\end{stanza}
\begin{stanza}
Wakhalela mn'Ethandaza;\\
Ethwele izono zam\\
Ngobusukw'eGetsemane,\\
Weva iminzunzu ngam.
\end{stanza}
\begin{stanza}
Wathob'intloko ngenxa yam\\
Ebethelelwe, Ejinga.\\
Kwampompoz'umsing'obomvu\\
Wafela mna nuMsindisi?
\end{stanza}
\end{hymn}

\begin{hymn}{43}{UYesu dlula ngapha}{\#86}
\begin{stanza}
Ukhon'olindel'uYesu?\\
Ofuna uxolelo?\\
Yiva iindab'ezimnandi;\\
UYesu dlula ngapha.
\end{stanza}
\begin{chorus}
\setchorusline{UYesu dlula ngapha,}
UYesu dlula ngapha,\\
Ngapha,namhlanje;\\
UYesu dlula ngapha\\
Udlula ngapha namhla.
\end{chorus}
\begin{stanza}
Ukhon'olahlekileyo?\\
Yiza nomthwalo wakho.\\
Ngenceba Uyakubiza.\\
UYesu dlula ngapha.
\end{stanza}
\begin{stanza}
Ukhon'ophuk'intliziyo,\\
Ofuna ukuphumla?\\
Phuml'ezingalweni zaKhe.\\
Wayam'esifubeni.
\end{stanza}
\begin{stanza}
Yiza kuMsindisi wakho.\\
Yiva uthando lwaKhe.\\
Ngena ngesango eliya;\\
Ekhaya ngaphezulu.
\end{stanza}
\end{hymn}

\begin{hymn}{10}{Jesus is our only message}{}

Mutu wa uthengawu\\
Yesu mu nthawi zonsezo\\
Tione Yesu yekha.
\end{stanza}
\begin{chorus}
\setchorusline{Ndiye nyimbo yathu yekha}
Ndiye nyimbo yathu yekha\\
Yesu nthawi zonsezo\\
Mombolo wamatamando\\
Mpulumutsi wathu 'yo.
\end{chorus}
\begin{stanza}
Yesu ndi mpulumutsi 'yo\\
Anatiwombola 'fe\\
Anatipatsa 'fe mphamvu\\
Wakutilungamitsa.
\end{stanza}
\begin{stanza}
Yesu ndiye mphamvu yathu\\
Amakhala mwa ife\\
Ndipo tisachite mantha\\
Tilimbikitsidwa 'ye.
\end{stanza}
\begin{stanza}
Ambuye wathu ndi Yesu\\
Chimwemwe kumvera 'ye.\\
Tidzipereke kwa Iye\\
Ndi mtima wathu wonse.
\end{stanza}
\end{hymn}
